    apartment_id varchar(10) PRIMARY KEY,

    apartment_name varchar(40) NOT NULL,

    no_flats int NOT NULL,

    create table Flat(

    • create an constraints table of designation by using below data
  • des_id - make sure it should not be duplicate and should not be empty (hint: use "PRIMARY KEY CONSTRAINT")
  • des_name - It should not be empty (hint: use "NOT NULL ,UNIQUE CONSTRAINTs")

    • create table employee(

    emp_id int PRIMARY KEY,

    emp_name varchar(50) NOT NULL,

    gender varchar(10) CHECK(gender in('male', 'female')),

    salary decimal(8,2) CHECK(salary >=10000 AND salary <=120000),

    experience decimal(3,2) CHECK(experience >=0 ),

    des_id varchar(10),

    Foreign key(des_id) references designation(des_id)

    );

    create table designation(

    des_id varchar(10) Primary key,

    des_name varChar(100) NOT NULL unique,

    department varchar(50) CHECK(department in('Technical','Non Technical', 'Finance', 'Support'))

    );
