    • CREATE TABLE Employee (

    emp_id int PRIMARY KEY,

    emp_name varchar(255) NOT NULL,

    salary decimal(10,2) CHECK(salary >=5000 AND salary <=100000),

    joining_date date DEFAULT(current_date)

    );

    create table Laptop (

    lap_id int primary Key,

    lap_name varchar(30) NOT NULL,

    lap_model varchar(30) NOT NULL,

    lap_warranty date NOT NULL,

    emp_id int,

    FOREIGN KEY (emp_id) REFERENCES employee(emp_id)

    );

    • create table Country(

    country_name varchar(30) PRIMARY KEY,

    population bigint NOT NULL,

    largest_river varchar(20) NOT NULL,

    main_crop varchar(10) NOT NULL,)

    isSeaShore boolean NOT NULL

    );

    create table Capital(

    capital_name varchar(20) PRIMARY KEY,

    population bigint NOT NULL,

    isSeaShore boolean NOT NULL,

    area varchar(50) NOT NULL,

    country_name varchar(30),

    FOREIGN KEY (country_name) REFERENCES country(country_name)

    );
