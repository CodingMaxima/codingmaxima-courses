create an constraints table of employee by using below data
  • id - make sure it should not be duplicate and should not be empty (hint: use "PRIMARY KEY CONSTRAINT")
  • first_name - it should not be empty (hint: use "NOT NULL CONSTRAINT")
  • last_name - it should not be empty (hint: use "NOT NULL CONSTRAINT")
  • gender - it should be either male or female or unknown(hint: use "CHECK CONSTRAINT")
  • designation - keep a fixed value for it. If you doesn't want to give an value for this particular column/for auto generating of the value without giving any manual values(hint: use "DEFAULT CONSTRAINT")
  • company_name - it should not be empty (hint: use "NOT NULL CONSTRAINT")
  • salary - it should be in between greater than or equal to 5000 and less than or equal to 100000(hint: use "CHECK CONSTRAINT")
  • age - it should be in between greater than or equal to 18 and less than or equal to 58(hint: use "CHECK CONSTRAINT")
  • bankaccountnumber - it should not be duplicate(hint: use "UNIQUE CONSTRAINT")
  • primaryphonenumber - keep a fixed value for it. If you doesn't want to give an value for this particular column/for auto generating of the value without giving any manual values(hint: use "DEFAULT CONSTRAINT")
  • employee_type - it should not be empty (hint: use "NOT NULL CONSTRAINT")
  • joining_date - keep a fixed value for it. If you doesn't want to give an value for this particular column/for auto generating of the value without giving any manual values(hint: use "DEFAULT CONSTRAINT")

    • CREATE TABLE employee (

    id int PRIMARY KEY,

    first_name varchar(255) NOT NULL,

    last_name varchar(255) NOT NULL,

    gender varchar(20) CHECK(gender in('male','female','unknown')),

    designation varchar(100) default 'bench',

    company_name varchar(255) NOT NULL,

    salary decimal(10,2) CHECK(salary >=5000 AND salary <=100000),

    age int CHECK(age >=18 AND age <=58),

    bankaccountnumber varchar(255) UNIQUE,

    primaryphonenumber bigint DEFAULT 0,

    employee_type varchar(100) DEFAULT 'full time',

    joining_date date DEFAULT(current_date)

    );
