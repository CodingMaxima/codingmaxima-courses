Pre-requisite data for alter & drop

Employee Table Student Table Movies Table

CREATE TABLE employee (

id bigint,

first_name varchar(100),

middle_name varchar(100),

last_name varchar(100),

designation varchar(100),

salary decimal(10,2),

gender varchar(20),

age int,

isactive boolean,

companyname varchar(100),

primaryphonenumber bigint,

secondaryphonenumber bigint

);

CREATE TABLE student (

id long,

first_name varchar(100),

middle_name varchar(100),

last_name varchar(100),

gender varchar(100),

age int,

aadharNumber long,

percentage decimal(4,2),

collegeName varchar(100),

grade varchar(5),

branch varchar(10)

);

CREATE TABLE movies (

movieid bigint,

moviename varchar(100),

genre varchar(100),

type varchar(100),

actor varchar(100),

language varchar(100),

contry varchar(100),

IMDBrating float(20),

picturequality varchar (100),

);
