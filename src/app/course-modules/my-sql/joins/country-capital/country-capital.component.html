Question-1

Print all the country details along with capital details which are having seashore(inner Join)

{{col}}
{{col}}

select * from country co inner join capital ca on co.country_name = ca.country_name where co.isSeaShore = true and ca.isSeaShore = true;

Question-2

Print all the country and capital details where only the countries contains sea shore but not the capital contains seashore(left join)

{{col}}
{{col}}

select * from country co inner join capital ca on co.country_name = ca.country_name where co.isSeaShore = true and ca.isSeaShore = false;

Question-3

Print the both country and capital details doesn't contains seashores

{{col}}
{{col}}

select * from country co inner join capital ca on co.country_name = ca.country_name where co.isSeaShore = false and ca.isSeaShore = false;

Question-4

print the country name ,country popuation and capital name, capital population by country population order(descending order)

{{col}}
{{col}}

select co.country_name, co.population, ca.capital_name, ca.population from country co inner join capital ca on co.country_name = ca.country_name order by co.population desc;
