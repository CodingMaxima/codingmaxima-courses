Question-1

Print all the availbale flats in the apartment along with apartment details

{{col}}
{{col}}

select * from flat f inner join apartment a on f.apartment_id = a.apartment_id where f.isAvailable = true;

Question-2

Print all the east facing appartment details along with the flats

{{col}}
{{col}}

select * from apartment a inner join flat f on a.apartment_id = f.apartment_id where a.facing = 'east';

Question-3

print all the apartment and flats details based on bedrooms order by ascending order

{{col}}
{{col}}

select * from flat f inner join apartment a on f.apartment_id = a.apartment_id order by f.no_bedrooms;

Question-4

Print 2 bedrooms available flats details where the apartment has lift facility.

{{col}}
{{col}}

select * from apartment a inner join flat f on f.apartment_id = a.apartment_id where f.isAvailable = true and f.no_bedrooms = 2 and a.isLift = true;

Question-5

print the west facing available flats which are in below 3rd floor order by floors.

{{col}}
{{col}}

select * from flat f inner join apartment a on f.apartment_id = a.apartment_id where f.facing = 'west' and f.isAvailable = true and f.floor < 3 order by f.floor;
