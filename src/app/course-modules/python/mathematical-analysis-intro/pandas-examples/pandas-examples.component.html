Reading and Exploring Data
import pandas as pd

# Read a CSV file into a DataFrame
data = pd.read_csv('data.csv')

# Display the first few rows of the DataFrame
print(data.head())

# Get summary statistics of the DataFrame
print(data.describe())

# Count the occurrences of each value in a column
print(data['category'].value_counts())

# Filter rows based on a condition
filtered_data = data[data['value'] > 100]

# Group data by a column and calculate the average
average_by_category = data.groupby('category')['value'].mean()
print(average_by_category)
Data Cleaning and Transformation
import pandas as pd

# Remove duplicates from a DataFrame
data.drop_duplicates(inplace=True)

# Handle missing values by filling them with the mean
data['age'].fillna(data['age'].mean(), inplace=True)

# Convert data types of columns
data['date'] = pd.to_datetime(data['date'])

# Apply a function to a column
data['category'] = data['category'].apply(lambda x: x.upper())

# Create a new column based on existing columns
data['total_sales'] = data['quantity'] * data['price']

Data Visualization
import pandas as pd
import matplotlib.pyplot as plt

# Plot a line chart of a time series
data.plot(x='date', y='value', kind='line')
plt.xlabel('Date')
plt.ylabel('Value')
plt.title('Value over Time')
plt.show()

# Create a bar chart of categorical data
data['category'].value_counts().plot(kind='bar')
plt.xlabel('Category')
plt.ylabel('Count')
plt.title('Category Counts')
plt.show()

# Generate a scatter plot
data.plot(x='age', y='income', kind='scatter')
plt.xlabel('Age')
plt.ylabel('Income')
plt.title('Income vs Age')
plt.show()
